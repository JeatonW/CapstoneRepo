CTRL_L + ALT_L + F::

	active = true
	tabs = 0

	py:
		paste("for i in range(0,0):")
		startCursor(0,4)
		highlight(1)
	java:
		paste("for(int i=START; i<SIZE; i+=INCREMENT) {\n\n}")
		startCursor(0,10)
		highlight(5)

	key(M1, 1):
		exit

	key(TAB, 1):
		py:
			moveCursor(0,10)
			highlight(1)
		java:
			moveCursor(0,4)
			highlight(4)

	key(TAB, 2):
		py:
			moveCursor(0,1)
			highlight(1)
		java:
			moveCursor(0,5)
			highlight(9)

	key(TAB, 3):
		moveCursor(1,0)
		paste("\t")
		active = false

CTRL_L + ALT_L + W::

	active = true

	py:
		paste("while(i < SIZE):")
	java:
		paste("while(i < SIZE) {\n\n}")

	startCursor(0, 10)
	highlight(4)


	while(active):

		if(mouseClicked):
			active = false

		if(tabPressed):
			moveCursor(1, 0)
			paste("\t")
			active = false
